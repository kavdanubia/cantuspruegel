\subsection*{Dir öffnet sich jetzt uns're Brust}
\index{Dir öffnet sich jetzt uns're Brust}%

\weise{Stimmt an mit hellem, hohem Klang}

\thestrophe Dir öffnet sich jetzt uns're Brust zu treuem Bruderleben, \\
\wdh{du teilest mit uns Leid und Lust in ernstem Geistesstreben.}

\thestrophe Sei fromm im Denken, Wort und Tat und wahre reine Sitten; \\
\wdh{das Böse, wann und wo es naht, sei immer kühn bestritten.}

\thestrophe Sei frisch und voller Jugendkraft in deinen Frühlingstagen, \\
\wdh{nur tatenfrohe Ritterschaft darf Siegerkronen tragen.} 

\thestrophe Sei frei im Geiste immerdar, das ist der echte Adel. \\
\wdh{Der freie Mann, der ist auch wahr und stark und sonder Tadel.}

\thestrophe Sei fröhlich, dass du jederzeit ein Liedlein könntest singen; \\
\wdh{fürwahr, ein trefflich gut Geleit in allen ernsten Dingen!}

\thestrophe So reich uns deine Bruderhand zum treuen Herzensbunde! \\
\wdh{Um uns schließt sich ein Bruderband durch diese Weihestunde.}
