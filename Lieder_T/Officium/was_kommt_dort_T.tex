\subsection*{Was kommt dort von der Höh'}
\index{Was kommt dort von der Höh'}%
\index{Reception}%

\weise{Vor der Reception}

\thestrophe \wdh{Was kommt dort von der Höh,} was kommt dort von der \\
ledernen Höh', ca, ca, ledernen Höh', was kommt dort von der Höh'?

\thestrophe \wdh{Es ist der Fuchsmajor,} es ist der lederne Fuchsmajor, \\
ca, ca, Fuchsmajor, es ist der Fuchsmajor.

\thestrophe \wdh{Was bringt der Fuchsmajor,} was bringt der lederne \\
Fuchsmajor, ca, ca, Fuchsmajor, was bringt der Fuchsmajor ?

\thestrophe \wdh{Er bringt uns seine Füchs',} er bringt uns seine ledernen Füchs', \\
ca, ca, ledernen Füchs', er bringt uns seine Füchs'.

\thestrophe \textit{FM-Solo}  \wdh{Ihr Diener, meine Herr'n,} Ihr Diener, meine \\
hohen Herr'n, ca, ca, hohen Herr'n, Ihr Diener, meine Herrn!

\thestrophe \textit{FM-Solo}  \wdh{Ich bring' Euch meine Füchs',} ich bring' Euch \\
meine ledern' Füchs', ca, ca, ledern' Füchs', ich bring Euch meine Füchs.}

\weise{Nach der Reception}

\thestrophe \wdh{So wird der Fink ein Fuchs,} so wird der lederne Fink ein Fuchs, \\
ca, ca, Fink ein Fuchs, so wird der Fink ein Fuchs.
