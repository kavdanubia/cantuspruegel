\subsection*{Wir hatten gebauet}
\index{Wir hatten gebauet}%

\comment{Nach der Ehrenbandverleihung}

\thestrophe Wir hatten gebauet ein stattliches Haus \\
\wdh{und drin auf Gott vertrauet trotz Wetter, Sturm und Graus.}

\thestrophe Wir lebten so traulich, so innig, so frei, \\
\wdh{den Schlechten ward es graulich, wir lebten gar zu treu!}

\thestrophe Sie lugten, sie suchten nach Trug und Verrat, \\
\wdh{verleumdeten, verfluchten die junge grüne Saat!}

\thestrophe Was Gott in uns legte, die Welt hat's veracht't, \\
\wdh{die Einigkeit erregte bei Guten selbst Verdacht!}

\thestrophe Man schalt es Verbrechen, man täuschte sich sehr; \\
\wdh{die Form kann man zerbrechen, die Liebe nimmermehr.}

\thestrophe Die Form ist zerbrochen, von außen herein, \\
\wdh{doch, was man drin gerochen, war eitel Dunst und Schein.}

\thestrophe Das Band ist zerschnitten, war Schwarz, Rot und Gold, \\
\wdh{und Gott hat es gelitten, wer weiß, was er gewollt!}

\thestrophe Das Haus mag zerfallen -was hat's dann für Not? \\
\wdh{Der Geist lebt in uns allen, und unsre Burg ist Gott!}
