\subsection*{Im Krug zum grünen Kranze}
\index{Im Krug zum grünen Kranze}%

\thestrophe Im Krug zum grünen Kranze, da kehrt' ich durstig ein; \\
\wdh{da saß ein Wand'rer drinnen, ja drinnen, am Tisch bei kühlem Wein.}

\thestrophe Ein Glas war eingegossen, das wurde nimmer leer; \\
\wdh{sein Haupt ruht' auf dem Bündel, ja Bündel, als wär's ihm viel zu schwer.}

\thestrophe Ich tät mich zu ihm setzen, ich sah ihm ins Gesicht, \\
\wdh{das schien mir gar befreundet, befreundet, und dennoch kannt' ich's nicht.}

\thestrophe Da sah auch mir ins Auge der fremde Wandersmann \\
\wdh{und füllte meinen Becher, ja Becher, und sah mich wieder an.}

\thestrophe Hei, was die Becher klangen, wie brannte Hand in Hand: \\
\wdh{"Es lebe die Liebste deine, ja deine, Herzbruder, im Vaterland!"}
