\subsection*{Ein Heller und ein Batzen}
\index{Ein Heller und ein Batzen}%

\thestrophe Ein Heller und ein Batzen, die waren beide mein, ja mein, \\
\wdh{der Heller ward zu Wasser, der Batzen ward zu Wein!} \\
\wdh{ \wdh{Heidi, heida, heida,} heidi, heida hei-da-ha-ha-ha-ha-ha-ha-ha}

\thestrophe Die Mädel und die Wirtsleut', die rufen beid': \wdh{o weh!} \\
\wdh{Die Wirtsleut', wenn ich komme, die Mädel, wenn ich geh', ja geh'.}

\thestrophe Mein' Stiefel sind zerrissen, mein' Schuh', die sind \wdh{entzwei:}, \\
\wdh{und draußen auf der Heide, da singt der Vogel frei, ja frei.}

\thestrophe Und gäb's kein Landstraß' nirgend, so blieb' ich still \wdh{zu Haus}, \\
\wdh{und gäb's kein Loch im Fasse, so tränk' ich gar nicht draus, ja draus.}

\thestrophe Das war 'ne rechte Freude, als mich der Herrgott schuf, ja schuf, \\
\wdh{'n Kerl wie Samt und Seide, nur schade, daß er suff, ja suff.}
